pipeline {
    agent {
        docker {
            image 'node:18-windowsservercore'
            args '--isolation=hyperv'  // or '--isolation=process' for Windows Server 2019+
        }
    }
    stages {
        stage('Build') {
            steps {
                bat 'npm install'
                bat 'npm run build'
            }
        }
        stage('Test') {
            steps {
                bat 'npm test'
            }
        }
    }
}
